if (this.prestationForm.valid) {
    const prestationData = this.prestationForm.value;

    if (this.editMode && this.selectedPrestation?.id) {
        // Inclure l'ID dans les données de la prestation à mettre à jour
        prestationData.id = this.selectedPrestation.id;

        // Mise à jour de la prestation
        this.prestationService.updateTPrestation(prestationData.id, prestationData).subscribe(() => {
            this.getAllPrestations();
            this.displayDialog = false;
        });
    } else {
        // Création d'une nouvelle prestation
        this.prestationService.createTPrestation(prestationData).subscribe(() => {
            this.getAllPrestations();
            this.displayDialog = false;
        });
    }
} else {
    this.prestationForm.markAllAsTouched();
}
